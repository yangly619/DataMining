1 Introduccin

El objetivo del trabajo es buscar una firma gentica que sirva para discriminar pacientes diagnosticados con tumor de prstata.
Los tumores de prstata se encuentran entre los cnceres ms heterogneos, tanto histolgicamente como clnicamente. El anlisis de expresin de microarrays se utiliz para determinar si las diferencias biolgicas globales subyacen a las caractersticas patolgicas comunes del cncer de prstata e identificar genes que podran anticipar el comportamiento clnico de esta enfermedad. Si bien no se encontr una correlacin de la expresin de la edad, se encontraron antgenos sricos especficos de la prstata (PSA) y medidas de invasin local, se identific un conjunto de genes que se correlacionaron fuertemente con el estado de diferenciacin tumoral segn lo medido por la puntuacin de Gleason. Adems, un modelo que utiliza solo datos de expresin gnica predijo con precisin el resultado del paciente despus de la prostatectoma. Estos resultados apoyan la idea de que el comportamiento clnico del cncer de prstata est vinculado a las diferencias subyacentes en la expresin de los genes que son detectables en el momento del diagnstico.

2 Metodologa

2.1 Datos

-136 muestras

2.2 Algoritmos de clasificacin

-Filtro01: gain.ratio
-Filtro02: symmetrical.uncertainty
-Filtro03: chi.squared
-Wrapper SFS

A partir de las variables seleccionadas con diferentes filtros, utilizaremos el mtodo wrapper SFS (Stepwise Forward Seletion) para busacar la mejor combinacin de variables para diferentes modelos (regresin logstica, ANN, SVM y rboles de decisin). Se usar el rea bajo la curva ROC como mtrica de evaluacin del rendimiento del modelo.
Para la validacin interna de todos los modelos usamos 10-fold cross-validation.

3.3 Stepwise Forward Selection

El clasificador de wrapper empaquetando algoritmos especficos de clasificacin / regresin y usando algunos algoritmos de optimizacin, elige la mejor combinacin de los atributos. Pero el tiempo de ejecucin es elevado, especialmente cuando tenemos 12,600 genes en nuestro conjunto de datos, tomara mucho tiempo en seleccionar la mejor combinacion de todas las combinaciones de genes posibles, por lo que en este trabajos usamos el algoritmo Stepwise Forward realiza la seleccin de variables entre las variables filtradas de arriba.

3.3.1 Funciones para ajuste de parmetros para regresin logstica, nnet, svm, dt.

3.3.1.1 ANN

addq <- function(x) paste0("`", x, "`") 
parameter.nnet <- function(d.train,d.test,variables){
   formula1 <- as.formula( paste("(Class=='Tumor')*1.0",paste(addq(variables),collapse = "+"),sep = "~") )
       m <- expand.grid(size=seq(1,6,by=1),decay=seq(0,6e-05,by=1e-05))
       auc <- vector()
       for (i in 1:dim(m)[1]) {
        set.seed(5)
        modelo <-  nnet(formula1,data=d.train,size=m$size[i],maxit=500, decay=m$decay[i],trace=FALSE)
        predict.test <- predict(modelo,d.test,type="raw")
        auc.test <- auc(d.test$Class,predict.test)[1]
        auc <- append(auc,auc.test)
       } 
       m$auc <- auc
  return(m)
}

3.3.1.2 SVM

parameter.svm <- function(d.train,d.test,variables){
       m <- expand.grid(cost=seq(100,800,by=100),gamma=10^(-6:-1))
       formula1 <- as.formula( paste("(Class=='Tumor')*1.0",paste(addq(variables),collapse = "+"),sep = "~") )
      auc <- vector()
      for (i in 1:dim(m)[1]) {
        modelo <-  svm(formula1,data=d.train,cost=m$cost[i],gamma=m$gamma[i],probability=TRUE)
        predict.test <- predict(modelo,d.test,probability=TRUE)
        auc.test <- auc(d.test$Class,predict.test)[1]
        auc <- append(auc,auc.test)
      }
   m$auc <- auc
  return(m )
}

3.3.1.3 DT

parameter.dt <- function(d.train,d.test,variables){
      cp <- c(0.01,0.001,0.0001,0.00001)
      formula1 <- as.formula( paste("(Class=='Tumor')*1.0",paste(addq(variables),collapse = "+"),sep = "~") )
      auc <- vector()
      for (i in 1:4) {
        dt.fit <- rpart(formula1,data=d.train,control=rpart.control(cp=cp[i]))
        dt.pred <- predict(dt.fit,d.test)
        auc.test <- auc(d.test$Class,dt.pred)[1]
        auc <- append(auc,auc.test)
      }
      r <- cbind(cp,auc)
  return( as.data.frame(r) )
}

3.3.2 Funcin para calcular la mtrica: AUC

estimaMetrica <- function(d.train,d.test,variables,modelo,parameters){
  formula <- as.formula( paste("(Class=='Tumor')*1.0",paste(addq(variables),collapse = "+"),sep = "~") )
      if(modelo=="nnet"){
        modelo <- nnet(formula,data = d.train,size=parameters$size,maxit=1000,decay=parameters$decay,trace=FALSE)
        predict.test <- predict(modelo,d.test,type="raw")
        auc.test <- auc(d.test$Class,predict.test)
      } 
      if(modelo=="svm"){
        modelo <-  svm(formula,data=d.train,size=parameters$cost,gamma=parameters$gamma,probability=TRUE)
        predict.test <- predict(modelo,d.test,probability=TRUE)
        auc.test <- auc(d.test$Class,predict.test)
      } 
      if(modelo=="dt"){
        formula.dt <- as.formula(paste("Class",paste(addq(variables),collapse = "+"),sep = "~") )
        dt.fit <- rpart(formula.dt, data=d.train, control=rpart.control(cp=parameters$cp))
        dt.pred <- predict(dt.fit,d.test)[,"Tumor"]
        auc.test <- auc(d.test$Class,dt.pred)
      }
     if(modelo=="glm"){
        modelo <- glm(formula,data=d.train,family = binomial("logit"))
        predict.test <- predict(modelo,d.test,type="response")
        auc.test <- auc(d.test$Class,predict.test)
        }
      return(auc.test)
}

3.3.3 Eval function for wrapper

Implementacin de eval.fun para el forward.search(attributes, eval.fun),se emplea 10-fold cross-validation para cada modelo:

k_fold <- function(subset,type){
  folds <- createFolds(data$Class,k=10)
  df <- vector()
  for (i in 1:10) {
   d.test <- data[folds[[i]],c(subset,"Class")]
   d.train <- data[-folds[[i]],c(subset,"Class")]
   
     if(type == "nnet"){
       parameter <- parameter.nnet(d.train,d.test,subset)
       best.pt <- parameter[which.max(parameter$auc),]
       result <- estimaMetrica(d.train,d.test,subset,"nnet",best.pt)
       }
     if(type == "svm"){
       parameter <- parameter.svm(d.train,d.test,subset)
       best.pt <- parameter[which.max(parameter$auc),]
       result <- estimaMetrica(d.train,d.test,subset,"svm",best.pt)}
     if(type == "dt"){
       parameter <- parameter.dt(d.train,d.test,subset)
       best.pt <- parameter[which.max(parameter$auc),]
       result <- estimaMetrica(d.train,d.test,subset,"dt",best.pt)}
     if(type =="glm"){
       result <- estimaMetrica(d.train,d.test,subset,"glm","p")}
   
   df <- cbind(df,result)
  }  
 return(df)
}
wraper.k_fold.nnet <- function(subset){
   m = k_fold(subset,"nnet")
   return(mean(m))
}

wraper.k_fold.svm <- function(subset){
 m = k_fold(subset,"svm")
   return(mean(m))
}

wraper.k_fold.dt <- function(subset){
 m = k_fold(subset,"dt")
   return(mean(m))
}
wraper.k_fold.glm <- function(subset){
 m = k_fold(subset,"glm")
   return(mean(m))
}
